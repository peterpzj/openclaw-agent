```python
import matplotlib.pyplot as plt
import matplotlib
matplotlib.use('Agg')
plt.rcParams['font.sans-serif'] = ['SimHei', 'Arial Unicode MS']
plt.rcParams['axes.unicode_minus'] = False

def create_bar_chart(data, labels, title, color='#4472C4'):
    """生成柱状图"""
    fig, ax = plt.subplots(figsize=(10, 6))
    bars = ax.bar(labels, data, color=color)
    ax.set_title(title, fontsize=16, fontweight='bold')
    for bar, value in zip(bars, data):
        ax.text(bar.get_x() + bar.get_width()/2., height, f'{value:.0f}',
                ha='center', va='bottom', fontsize=11)
    plt.tight_layout()
    output_path = f'/tmp/chart_{title}.png'
    plt.savefig(output_path, dpi=150, bbox_inches='tight')
    plt.close()
    return output_path

def create_pie_chart(sizes, labels, title, colors=None):
    """生成饼图"""
    fig, ax = plt.subplots(figsize=(8, 8))
    ax.pie(sizes, labels=labels, autopct='%1.1f%%', colors=colors,
           textprops={'fontsize': 12})
    ax.set_title(title, fontsize=16, fontweight='bold')
    plt.tight_layout()
    output_path = f'/tmp/chart_{title}.png'
    plt.savefig(output_path, dpi=150, bbox_inches='tight')
    plt.close()
    return output_path

def create_line_chart(x, y, title, labels=['x', 'y']):
    """生成折线图"""
    fig, ax = plt.subplots(figsize=(10, 6))
    ax.plot(x, y, marker='o', linewidth=2, markersize=8)
    ax.set_xlabel(labels[0], fontsize=12)
    ax.set_ylabel(labels[1], fontsize=12)
    ax.set_title(title, fontsize=16, fontweight='bold')
    ax.grid(True, linestyle='--', alpha=0.7)
    plt.tight_layout()
    output_path = f'/tmp/chart_{title}.png'
    plt.savefig(output_path, dpi=150, bbox_inches='tight')
    plt.close()
    return output_path
```

```python
# PPT文件提取
from pptx import Presentation
def extract_pptx(file_path):
    prs = Presentation(file_path)
    slides_data = []
    for i, slide in enumerate(prs.slides):
        title = ""
        content = []
        for shape in slide.shapes:
            if shape.has_text_frame:
                for para in shape.text_frame.paragraphs:
                    text = para.text.strip()
                    if text:
                        if not title:
                            title = text
                        else:
                            content.append(text)
        slides_data.append({'title': title, 'content': content})
    return slides_data
```

## 📊 PPT页面生成代码模板

```python
from pptx import Presentation
from pptx.util import Inches, Pt, Cm
from pptx.dml.color import RGBColor
from pptx.enum.text import PP_ALIGN
from pptx.enum.shapes import MSO_SHAPE
import os

# ==================== 初始化 ====================
prs = Presentation()
prs.slide_width = Inches(13.333)  # 16:9
prs.slide_height = Inches(7.5)

def set_fill(shape, color):
    """设置填充色"""
    shape.fill.solid()
    shape.fill.fore_color.rgb = color

def add_textbox(slide, left, top, width, height, text, size=18, bold=False, color=RGBColor(33,33,33), align=PP_ALIGN.LEFT):
    """添加文本框"""
    txBox = slide.shapes.add_textbox(Inches(left), Inches(top), Inches(width), Inches(height))
    tf = txBox.text_frame
    p = tf.paragraphs[0]
    p.text = text
    p.font.size = Pt(size)
    p.font.bold = bold
    p.font.color.rgb = color
    p.alignment = align
    return txBox

# ==================== 封面页 ====================
def create_cover(title, subtitle="", author="", date="", style="商务专业"):
    slide = prs.slides.add_slide(prs.slide_layouts[6])
    colors = STYLES[style]
    
    # 背景
    bg = slide.shapes.add_shape(MSO_SHAPE.RECTANGLE, 0, 0, prs.slide_width, prs.slide_height)
    set_fill(bg, colors["bg_light"])
    
    # 顶部装饰条
    top_bar = slide.shapes.add_shape(MSO_SHAPE.RECTANGLE, 0, 0, prs.slide_width, Inches(0.3))
    set_fill(top_bar, colors["primary"])
    
    # 主标题
    add_textbox(slide, 0.8, 2.5, 11.7, 1.5, title, size=48, bold=True, 
                color=colors["primary"], align=PP_ALIGN.CENTER)
    
    # 副标题
    if subtitle:
        add_textbox(slide, 0.8, 4.0, 11.7, 0.8, subtitle, size=24,
                    color=colors["secondary"], align=PP_ALIGN.CENTER)
    
    # 作者和日期
    info = f"{author}  |  {date}" if author or date else ""
    if info:
        add_textbox(slide, 0.8, 5.5, 11.7, 0.5, info, size=14,
                    color=RGBColor(128,128,128), align=PP_ALIGN.CENTER)
    
    return slide

# ==================== 目录页 ====================
def create_toc(items, style="商务专业"):
    slide = prs.slides.add_slide(prs.slide_layouts[6])
    colors = STYLES[style]
    
    # 标题栏
    title_bar = slide.shapes.add_shape(MSO_SHAPE.RECTANGLE, 0, 0, prs.slide_width, Inches(1.2))
    set_fill(title_bar, colors["primary"])
    add_textbox(slide, 0.5, 0.35, 12, 0.8, "目录", size=32, bold=True,
                color=colors["text_light"], align=PP_ALIGN.LEFT)
    
    # 目录项
    y_start = 1.8
    for i, item in enumerate(items):
        # 序号圆圈
        circle = slide.shapes.add_shape(MSO_SHAPE.OVAL, Inches(1), Inches(y_start + i*0.9), Inches(0.5), Inches(0.5))
        set_fill(circle, colors["secondary"])
        add_textbox(slide, 1, y_start + i*0.9 + 0.08, 0.5, 0.4, str(i+1), size=18, bold=True,
                    color=colors["text_light"], align=PP_ALIGN.CENTER)
        
        # 目录文字
        add_textbox(slide, 1.8, y_start + i*0.9, 10, 0.5, item, size=22,
                    color=colors["text_dark"])
    
    return slide

# ==================== 内容页 ====================
def create_content_page(title, points, style="商务专业", image_path=None):
    slide = prs.slides.add_slide(prs.slide_layouts[6])
    colors = STYLES[style]
    
    # 标题栏
    title_bar = slide.shapes.add_shape(MSO_SHAPE.RECTANGLE, 0, 0, prs.slide_width, Inches(1.2))
    set_fill(title_bar, colors["primary"])
    add_textbox(slide, 0.5, 0.35, 12, 0.8, title, size=28, bold=True,
                color=colors["text_light"], align=PP_ALIGN.LEFT)
    
    # 内容区
    content_box = slide.shapes.add_textbox(Inches(0.8), Inches(1.8), Inches(11.5), Inches(5.2))
    tf = content_box.text_frame
    tf.word_wrap = True
    
    for i, point in enumerate(points):
        if i == 0:
            p = tf.paragraphs[0]
        else:
            p = tf.add_paragraph()
        p.text = f"• {point}"
        p.font.size = Pt(20)
        p.font.color.rgb = colors["text_dark"]
        p.space_after = Pt(14)
    
    # 配图（如果有）
    if image_path and os.path.exists(image_path):
        slide.shapes.add_picture(image_path, Inches(9.5), Inches(2), width=Inches(3.5))
    
    return slide

# ==================== 图表页 ====================
def create_chart_page(title, chart_image_path, style="商务专业"):
    slide = prs.slides.add_slide(prs.slide_layouts[6])
    colors = STYLES[style]
    
    # 标题栏
    title_bar = slide.shapes.add_shape(MSO_SHAPE.RECTANGLE, 0, 0, prs.slide_width, Inches(1.2))
    set_fill(title_bar, colors["primary"])
    add_textbox(slide, 0.5, 0.35, 12, 0.8, title, size=28, bold=True,
                color=colors["text_light"], align=PP_ALIGN.LEFT)
    
    # 图表
    if os.path.exists(chart_image_path):
        slide.shapes.add_picture(chart_image_path, Inches(1.5), Inches(1.8), width=Inches(10))
    
    return slide

# ==================== 总结页 ====================
def create_summary(title, points, style="商务专业"):
    slide = prs.slides.add_slide(prs.slide_layouts[6])
    colors = STYLES[style]
    
    # 标题栏
    title_bar = slide.shapes.add_shape(MSO_SHAPE.RECTANGLE, 0, 0, prs.slide_width, Inches(1.2))
    set_fill(title_bar, colors["secondary"])
    add_textbox(slide, 0.5, 0.35, 12, 0.8, title, size=28, bold=True,
                color=colors["text_light"], align=PP_ALIGN.LEFT)
    
    # 内容框
    box = slide.shapes.add_shape(MSO_SHAPE.ROUNDED_RECTANGLE, 
                                  Inches(0.8), Inches(1.6), Inches(11.7), Inches(5.2))
    set_fill(box, colors["bg_light"])
    box.line.color.rgb = colors["accent"]
    
    # 总结要点
    content_box = slide.shapes.add_textbox(Inches(1.2), Inches(2), Inches(11), Inches(4.5))
    tf = content_box.text_frame
    tf.word_wrap = True

        else:
            p = tf.add_paragraph()
        p.text = f"✓ {point}"
        p.font.size = Pt(22)
        p.font.color.rgb = colors["primary"]
        p.space_after = Pt(16)
    
    return slide

# ==================== 结束页 ====================
def create_end(message="谢谢观看", style="商务专业"):

                color=colors["text_light"], align=PP_ALIGN.CENTER)
    
    return slide
```
